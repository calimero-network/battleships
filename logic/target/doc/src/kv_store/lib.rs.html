1//! # Battleship Game Logic
2//!
3//! This crate implements a complete battleship game using the Calimero SDK.
4//! It provides a modular, well-documented implementation following Domain-Driven Design (DDD) principles.
5//!
6//! ## Architecture Overview
7//!
8//! The codebase is organized into several modules, each with a specific responsibility:
9//!
10//! - **`board`** - Board representation, coordinates, and cell types
11//! - **`ships`** - Ship definitions, fleet management, and ship validation
12//! - **`players`** - Player management and private board logic
13//! - **`game`** - Core game logic and match management
14//! - **`events`** - Game events and state changes
15//! - **`validation`** - Comprehensive validation strategy pattern implementation
16//!
17//! ## Key Features
18//!
19//! ### Validation Strategy Pattern
20//! The validation system uses a sophisticated Strategy Pattern implementation that provides:
21//! - **Extensibility**: Easy to add new validation rules
22//! - **Composability**: Mix and match validation strategies
23//! - **Testability**: Each strategy can be tested independently
24//! - **Maintainability**: Validation logic is organized and separated
25//!
26//! ### Game Flow
27//! 1. **Match Creation**: Players create matches and join them
28//! 2. **Ship Placement**: Players place their ships using coordinate strings
29//! 3. **Turn-based Gameplay**: Players take turns shooting at opponent boards
30//! 4. **Shot Resolution**: Shots are resolved and results are recorded
31//! 5. **Win Condition**: Game ends when all ships of one player are sunk
32//!
33//! ## Usage Examples
34//!
35//! ### Creating a Match
36//! ```rust
37//! use battleship::BattleshipState;
38//! 
39//! let mut state = BattleshipState::init();
40//! let match_id = state.create_match("player2_base58_key".to_string())?;
41//! ```
42//!
43//! ### Placing Ships
44//! ```rust
45//! let ships = vec![
46//!     "0,0;0,1;0,2".to_string(), // 3-length ship
47//!     "2,0;2,1;2,2;2,3".to_string(), // 4-length ship
48//!     // ... more ships
49//! ];
50//! state.place_ships(&match_id, ships)?;
51//! ```
52//!
53//! ### Taking Shots
54//! ```rust
55//! state.propose_shot(&match_id, 5, 3)?; // Shoot at (5,3)
56//! let result = state.acknowledge_shot(&match_id)?; // Resolve the shot
57//! ```
58//!
59//! ## Error Handling
60//!
61//! The game uses a comprehensive error system with specific error types:
62//! - `GameError::NotFound` - Resource not found
63//! - `GameError::Invalid` - Invalid input or state
64//! - `GameError::Forbidden` - Operation not allowed
65//! - `GameError::Finished` - Game has ended
66//!
67//! ## Documentation
68//!
69//! For detailed API documentation, run:
70//! ```bash
71//! cargo doc --open
72//! ```

167/// Comprehensive error type for all game operations
168///
169/// This enum represents all possible errors that can occur during game operations.
170/// It uses a tagged representation for JSON serialization, making it easy to
171/// handle different error types on the client side.
172///
173/// # Variants
174/// * `NotFound(String)` - Resource not found (e.g., match ID, player)
175/// * `Invalid(&'static str)` - Invalid input or state (e.g., invalid coordinates, game rules)
176/// * `Forbidden(&'static str)` - Operation not allowed (e.g., not your turn, not a player)
177/// * `Finished` - Game has already ended
178///
179/// # Example
180/// ```rust
181/// use battleship::GameError;
182///
183/// match result {
184///     Err(GameError::NotFound(id)) => println!("Match {} not found", id),
185///     Err(GameError::Invalid(msg)) => println!("Invalid operation: {}", msg),
186///     Err(GameError::Forbidden(msg)) => println!("Forbidden: {}", msg),
187///     Err(GameError::Finished) => println!("Game has ended"),
188///     Ok(_) => println!("Success!"),
189/// }
190/// ```
191#[derive(Debug, Error, Serialize)]
192#[serde(crate = "calimero_sdk::serde")]
193#[serde(tag = "kind", content = "data")]
194pub enum GameError {
195    /// Resource not found (e.g., match ID, player)
196    #[error("not found: {0}")]
197    NotFound(String),
198    /// Invalid input or state (e.g., invalid coordinates, game rules)
199    #[error("invalid input: {0}")]
200    Invalid(&'static str),
201    /// Operation not allowed (e.g., not your turn, not a player)
202    #[error("forbidden: {0}")]
203    Forbidden(&'static str),
204    /// Game has already ended
205    #[error("already finished")]
206    Finished,
207}

213/// Main application state for the battleship game
214///
215/// This struct holds the global state of the battleship game application,
216/// including the active match and metadata for ID generation. It implements
217/// the Calimero SDK's state management system.
218///
219/// # Fields
220/// * `id_nonce` - Counter for generating unique match IDs
221/// * `created_ms` - Timestamp when the state was created
222/// * `active_match` - Currently active match (if any)
223///
224/// # Example
225/// ```rust
226/// use battleship::BattleshipState;
227///
228/// let state = BattleshipState::init();
229/// let match_id = state.create_match("player2_key".to_string())?;
230/// ```
231#[app::state]
232#[derive(Debug, BorshSerialize, BorshDeserialize)]
233#[borsh(crate = "calimero_sdk::borsh")]
234pub struct BattleshipState {
235    /// Counter for generating unique match IDs
236    id_nonce: u64,
237    /// Timestamp when the state was created
238    created_ms: u64,
239    /// Currently active match (if any)
240    active_match: Option<Match>,
241}
242
243#[app::logic]
244impl BattleshipState {
245    #[app::init]
246    pub fn init() -> BattleshipState {
247        BattleshipState {
248            id_nonce: 0,
249            created_ms: env::time_now(),
250            active_match: None,
251        }
252    }
253
254    fn next_id(&mut self) -> String {
255        self.id_nonce = self.id_nonce.wrapping_add(1);
256        format!("match-{}-{}", env::time_now(), self.id_nonce)
257    }
258
259    fn get_active_match(&self) -> app::Result<&Match> {
260        self.active_match.as_ref()
261            .ok_or_else(|| calimero_sdk::types::Error::from(GameError::Invalid("no active match")))
262    }
263
264    fn get_active_match_mut(&mut self) -> app::Result<&mut Match> {
265        self.active_match.as_mut()
266            .ok_or_else(|| calimero_sdk::types::Error::from(GameError::Invalid("no active match")))
267    }
268}

274/// Public API for game operations
275///
276/// This implementation provides all the public methods for interacting with
277/// the battleship game, including match creation, ship placement, and gameplay.
278#[app::logic]
279impl BattleshipState {
280    /// Creates a new match between the current player and another player
281    ///
282    /// This method creates a new battleship match and sets up the initial
283    /// game state. Only one active match is allowed at a time.
284    ///
285    /// # Arguments
286    /// * `player2` - Base58-encoded public key of the second player
287    ///
288    /// # Returns
289    /// * `Ok(String)` - The unique match ID
290    /// * `Err(GameError)` - If another match is active or players are the same
291    ///
292    /// # Example
293    /// ```rust
294    /// let mut state = BattleshipState::init();
295    /// let match_id = state.create_match("player2_base58_key".to_string())?;
296    /// println!("Created match: {}", match_id);
297    /// ```
298    pub fn create_match(&mut self, player2: String) -> app::Result<String> {
299        if self.active_match.is_some() && !self.get_active_match()?.is_finished() {
300            app::bail!(GameError::Invalid("another match is active"));
301        }
302
303        let player1 = PublicKey::from_executor_id()?;
304        let player2_pk = PublicKey::from_base58(&player2)?;
305        
306        if player1 == player2_pk {
307            app::bail!(GameError::Invalid("players must differ"));
308        }
309
310        let id = self.next_id();
311        self.active_match = Some(Match::new(id.clone(), player1, player2_pk));
312        
313        app::emit!(Event::MatchCreated { id: &id });
314        Ok(id)
315}
316
317    /// Places ships on the current player's board
318    ///
319    /// This method allows a player to place their fleet of ships on their board.
320    /// Ships must follow the standard battleship rules: 1x5, 1x4, 2x3, 1x2 lengths.
321    ///
322    /// # Arguments
323    /// * `match_id` - The ID of the match
324    /// * `ships` - Vector of ship coordinate strings in format "x1,y1;x2,y2;..."
325    ///
326    /// # Returns
327    /// * `Ok(())` - Ships placed successfully
328    /// * `Err(GameError)` - If match not found, not a player, or invalid ship placement
329    ///
330    /// # Example
331    /// ```rust
332    /// let ships = vec![
333    ///     "0,0;0,1;0,2".to_string(), // 3-length ship
334    ///     "2,0;2,1;2,2;2,3".to_string(), // 4-length ship
335    ///     "5,5;5,6".to_string(), // 2-length ship
336    /// ];
337    /// state.place_ships(&match_id, ships)?;
338    /// ```
339    pub fn place_ships(&mut self, match_id: &str, ships: Vec<String>) -> app::Result<()> {
340        let match_state = self.get_active_match_mut()?;
341        if match_id != match_state.id {
342            app::bail!(GameError::NotFound(match_id.to_string()));
343        }
344        if match_state.is_finished() {
345            app::bail!(GameError::Finished);
346        }
347
348        let caller = PublicKey::from_executor_id()?;
349        if !match_state.is_player(&caller) {
350            app::bail!(GameError::Forbidden("not a player"));
351        }
352
353        let mut priv_boards = PrivateBoards::private_load_or_default()?;
354        let mut priv_mut = priv_boards.as_mut();
355        let key = PrivateBoards::key(match_id);
356        let mut pb = priv_mut.boards.get(&key)?.unwrap_or(PlayerBoard::new());
357        
358        pb.place_ships(ships)?;
359        priv_mut.boards.insert(key, pb)?;
360
361        // Update match state
362        if caller == match_state.player1 {
363            match_state.placed_p1 = true;
364        } else {
365            match_state.placed_p2 = true;
366        }
367
368        app::emit!(Event::ShipsPlaced { id: match_id });
369        Ok(())
370}

372    pub fn propose_shot(&mut self, match_id: &str, x: u8, y: u8) -> app::Result<()> {
373        let match_state = self.get_active_match_mut()?;
374        if match_id != match_state.id {
375            app::bail!(GameError::NotFound(match_id.to_string()));
376        }
377
378        let caller = PublicKey::from_executor_id()?;
379        match_state.propose_shot(caller, x, y)?;
380        
381        app::emit!(Event::ShotProposed { id: match_id, x, y });
382        Ok(())
383}

385    pub fn acknowledge_shot(&mut self, match_id: &str) -> app::Result<String> {
386        let match_state = self.get_active_match_mut()?;
387        if match_id != match_state.id {
388            app::bail!(GameError::NotFound(match_id.to_string()));
389        }
390        if match_state.is_finished() {
391            app::bail!(GameError::Finished);
392        }
393
394        let caller = PublicKey::from_executor_id()?;
395        match_state.acknowledge_shot(caller)?;
396
397        // Resolve the shot
398        let mut priv_boards = PrivateBoards::private_load_or_default()?;
399        let mut priv_mut = priv_boards.as_mut();
400        let key = PrivateBoards::key(match_id);
401        let mut target_pb = priv_mut.boards.get(&key)?
402            .ok_or_else(|| calimero_sdk::types::Error::from(GameError::Invalid("target board unavailable")))?;
403
404        let result = ShotResolver::resolve_shot(match_state, &mut target_pb)?;
405        priv_mut.boards.insert(key, target_pb)?;
406        
407        if match_state.winner.is_some() {
408            app::emit!(Event::Winner { id: match_id });
409            app::emit!(Event::MatchEnded { id: match_id });
410        }
411
412        app::emit!(Event::ShotFired { 
413            id: match_id, 
414            x: match_state.pending_x.unwrap_or(0),
415            y: match_state.pending_y.unwrap_or(0),
416            result: &result 
417        });
418
419        Ok(result)
420    }

422    pub fn get_own_board(&self, match_id: &str) -> app::Result<OwnBoardView> {
423        let match_state = self.get_active_match()?;
424        if match_id != match_state.id {
425            app::bail!(GameError::NotFound(match_id.to_string()));
426        }
427
428        let caller = PublicKey::from_executor_id()?;
429        let key = match_id.to_string();
430        let priv_boards = PrivateBoards::private_load_or_default()?;
431        let pb = priv_boards.boards.get(&key)?
432            .ok_or_else(|| calimero_sdk::types::Error::from(GameError::NotFound(match_id.to_string())))?;
433
434        let mut board = pb.get_board().0.clone();
435
436        // Add pending shot if targeting this player
437        if let Some(pending_target) = &match_state.pending_target {
438            if *pending_target == caller {
439                if let (Some(x), Some(y)) = (match_state.pending_x, match_state.pending_y) {
440                    let idx = (y as usize) * (BOARD_SIZE as usize) + (x as usize);
441                    if idx < board.len() {
442                        board[idx] = Cell::Pending.to_u8();
443                    }
444                }
445            }
446        }
447
448        Ok(OwnBoardView { size: BOARD_SIZE, board })
449    }

451    pub fn get_shots(&self, match_id: &str) -> app::Result<ShotsView> {
452        let match_state = self.get_active_match()?;
453        if match_id != match_state.id {
454            app::bail!(GameError::NotFound(match_id.to_string()));
455        }
456
457        let caller = PublicKey::from_executor_id()?;
458        if !match_state.is_player(&caller) {
459            app::bail!(GameError::Forbidden("not a player"));
460        }
461
462        let shots = match_state.get_shots_for_player(&caller);
463        Ok(ShotsView { size: BOARD_SIZE, shots: shots.0.clone() })
464    }

466    pub fn get_matches(&self) -> app::Result<Vec<String>> {
467        let priv_boards = PrivateBoards::private_load_or_default()?;
468        let ids: Vec<String> = priv_boards
469            .boards
470            .entries()?
471            .map(|(k, _)| k)
472            .collect();
473        Ok(ids)
474    }
