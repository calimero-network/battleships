1//! # Game Module
2//!
3//! This module contains the core game logic and match management functionality
4//! for the battleship game. It handles match creation, turn management, shot
5//! processing, and game state transitions.
6//!
7//! ## Key Types
8//!
9//! - **`Match`** - Represents a game match between two players
10//! - **`ShotResolver`** - Service for resolving shot outcomes
11//!
12//! ## Game Flow
13//!
14//! 1. **Match Creation**: Two players create a match
15//! 2. **Ship Placement**: Both players place their ships
16//! 3. **Turn-based Gameplay**: Players take turns shooting
17//! 4. **Shot Resolution**: Shots are resolved and results recorded
18//! 5. **Win Condition**: Game ends when all ships are sunk
19//!
20//! ## Match State
21//!
22//! Each match tracks:
23//! - Player information and turn order
24//! - Ship placement status for both players
25//! - Pending shots and their resolution
26//! - Shot history for both players
27//! - Winner determination
28//!
29//! ## Usage Examples
30//!
31//! ### Creating a Match
32//! ```rust
33//! use battleship::game::Match;
34//! use battleship::players::PublicKey;
35//!
36//! let player1 = PublicKey::from_executor_id()?;
37//! let player2 = PublicKey::from_base58("player2_key")?;
38//! let match_id = "match-123".to_string();
39//! let game = Match::new(match_id, player1, player2);
40//! ```
41//!
42//! ### Processing Shots
43//! ```rust
44//! use battleship::game::ShotResolver;
45//!
46//! let result = ShotResolver::resolve_shot(&mut match_state, &mut target_board)?;
47//! println!("Shot result: {}", result);
48//! ```

60/// Represents a game match between two players
61///
62/// A match contains all the state information for a battleship game, including
63/// player information, turn management, shot tracking, and game status. It
64/// handles the complete game flow from creation to completion.
65///
66/// # Fields
67/// * `id` - Unique identifier for the match
68/// * `player1` - First player's public key
69/// * `player2` - Second player's public key
70/// * `turn` - Current player whose turn it is
71/// * `winner` - Winner of the match (if any)
72/// * `placed_p1` - Whether player1 has placed their ships
73/// * `placed_p2` - Whether player2 has placed their ships
74/// * `pending_x` - X coordinate of pending shot
75/// * `pending_y` - Y coordinate of pending shot
76/// * `pending_shooter` - Player who fired the pending shot
77/// * `pending_target` - Player targeted by the pending shot
78/// * `shots_p1` - Player1's shot history board
79/// * `shots_p2` - Player2's shot history board
80///
81/// # Game States
82/// - **Setup**: Both players placing ships
83/// - **Active**: Turn-based gameplay with shots
84/// - **Pending**: Shot fired, waiting for resolution
85/// - **Finished**: Game completed with winner
86///
87/// # Example
88/// ```rust
89/// use battleship::game::Match;
90/// use battleship::players::PublicKey;
91///
92/// let player1 = PublicKey::from_executor_id()?;
93/// let player2 = PublicKey::from_base58("player2_key")?;
94/// let match_id = "match-123".to_string();
95/// let game = Match::new(match_id, player1, player2);
96/// assert_eq!(game.turn, player1);
97/// assert!(!game.is_finished());
98/// ```
99#[derive(Debug, Clone, BorshSerialize, BorshDeserialize, Serialize, Deserialize)]
100#[borsh(crate = "calimero_sdk::borsh")]
101#[serde(crate = "calimero_sdk::serde")]
102pub struct Match {
103    /// Unique identifier for the match
104    pub id: String,
105    /// First player's public key
106    pub player1: PublicKey,
107    /// Second player's public key
108    pub player2: PublicKey,
109    /// Current player whose turn it is
110    pub turn: PublicKey,
111    /// Winner of the match (if any)
112    pub winner: Option<PublicKey>,
113    /// Whether player1 has placed their ships
114    pub placed_p1: bool,
115    /// Whether player2 has placed their ships
116    pub placed_p2: bool,
117    /// X coordinate of pending shot
118    pub pending_x: Option<u8>,
119    /// Y coordinate of pending shot
120    pub pending_y: Option<u8>,
121    /// Player who fired the pending shot
122    pub pending_shooter: Option<PublicKey>,
123    /// Player targeted by the pending shot
124    pub pending_target: Option<PublicKey>,
125    /// Player1's shot history board
126    pub shots_p1: Board,
127    /// Player2's shot history board
128    pub shots_p2: Board,
129}
130
131impl Match {
132    pub fn new(id: String, player1: PublicKey, player2: PublicKey) -> Match {
133        Match {
134            id,
135            player1: player1.clone(),
136            player2: player2.clone(),
137            turn: player1,
138            winner: None,
139            placed_p1: false,
140            placed_p2: false,
141            pending_x: None,
142            pending_y: None,
143            pending_shooter: None,
144            pending_target: None,
145            shots_p1: Board::new_zeroed(BOARD_SIZE),
146            shots_p2: Board::new_zeroed(BOARD_SIZE),
147        }
148    }
149
150    pub fn is_player(&self, player: &PublicKey) -> bool {
151        *player == self.player1 || *player == self.player2
152    }
153
154    pub fn get_opponent(&self, player: &PublicKey) -> PublicKey {
155        if *player == self.player1 {
156            self.player2.clone()
157        } else {
158            self.player1.clone()
159        }
160    }
161
162    pub fn is_turn(&self, player: &PublicKey) -> bool {
163        self.turn == *player
164    }
165
166    pub fn switch_turn(&mut self) {
167        self.turn = self.get_opponent(&self.turn);
168    }
169
170    pub fn is_finished(&self) -> bool {
171        self.winner.is_some()
172    }
173
174    pub fn both_players_placed(&self) -> bool {
175        self.placed_p1 && self.placed_p2
176    }
177
178    pub fn has_pending_shot(&self) -> bool {
179        self.pending_x.is_some()
180    }
181
182    pub fn propose_shot(&mut self, shooter: PublicKey, x: u8, y: u8) -> Result<(), GameError> {
183        if self.is_finished() {
184            return Err(GameError::Finished);
185        }
186        if !self.both_players_placed() {
187            return Err(GameError::Invalid("both players must place ships first"));
188        }
189        if x >= BOARD_SIZE || y >= BOARD_SIZE {
190            return Err(GameError::Invalid("out of bounds"));
191        }
192        if self.has_pending_shot() {
193            return Err(GameError::Invalid("shot already pending"));
194        }
195        if !self.is_player(&shooter) {
196            return Err(GameError::Forbidden("not a player"));
197        }
198        if !self.is_turn(&shooter) {
199            return Err(GameError::Forbidden("not your turn"));
200        }
201
202        // Record shot in shooter's board
203        if shooter == self.player1 {
204            self.shots_p1.set(BOARD_SIZE, x, y, Cell::Pending);
205        } else {
206            self.shots_p2.set(BOARD_SIZE, x, y, Cell::Pending);
207        }
208
209        // Set pending shot
210        let opponent = self.get_opponent(&shooter);
211        self.pending_x = Some(x);
212        self.pending_y = Some(y);
213        self.pending_shooter = Some(shooter);
214        self.pending_target = Some(opponent);
215
216        Ok(())
217    }
218
219    pub fn acknowledge_shot(&mut self, target: PublicKey) -> Result<String, GameError> {
220        let pending_target = self.pending_target.as_ref()
221            .ok_or_else(|| GameError::Invalid("no pending shot"))?;
222
223        if *pending_target != target {
224            return Err(GameError::Forbidden("not the target"));
225        }
226
227        // This will be resolved by the shot resolver
228        Ok("acknowledged".to_string())
229    }
230
231    pub fn resolve_shot(&mut self, is_hit: bool) -> String {
232        let x = self.pending_x.take().expect("pending x should exist");
233        let y = self.pending_y.take().expect("pending y should exist");
234        let shooter = self.pending_shooter.take().expect("pending shooter should exist");
235        self.pending_target = None;
236
237        let result = if is_hit { "hit" } else { "miss" };
238
239        // Update shooter's board with result
240        let shot_result = if is_hit { Cell::Hit } else { Cell::Miss };
241        if shooter == self.player1 {
242            self.shots_p1.set(BOARD_SIZE, x, y, shot_result);
243        } else {
244            self.shots_p2.set(BOARD_SIZE, x, y, shot_result);
245        }
246
247        // Switch turns if no winner
248        if self.winner.is_none() {
249            self.switch_turn();
250        }
251
252        result.to_string()
253    }
254
255    pub fn set_winner(&mut self, winner: PublicKey) {
256        self.winner = Some(winner);
257    }
258
259    pub fn get_shots_for_player(&self, player: &PublicKey) -> &Board {
260        if *player == self.player1 {
261            &self.shots_p1
262        } else {
263            &self.shots_p2
264        }
265    }
266}

272/// Service for resolving shot outcomes in the game
273///
274/// The ShotResolver handles the logic for determining whether a shot hits or
275/// misses, updating the target board accordingly, and managing win conditions.
276/// It's responsible for the core game mechanics of shot resolution.
277///
278/// # Shot Resolution Process
279/// 1. Extract pending shot coordinates and shooter information
280/// 2. Check if the shot hits a ship on the target board
281/// 3. Update the target board with hit/miss result
282/// 4. Decrement ship count if hit
283/// 5. Set winner if all ships are sunk
284/// 6. Update match state with shot result
285///
286/// # Example
287/// ```rust
288/// use battleship::game::ShotResolver;
289///
290/// let result = ShotResolver::resolve_shot(&mut match_state, &mut target_board)?;
291/// match result.as_str() {
292///     "hit" => println!("Shot hit!"),
293///     "miss" => println!("Shot missed!"),
294///     _ => unreachable!(),
295/// }
296/// ```
297pub struct ShotResolver;
298
299impl ShotResolver {
300    pub fn resolve_shot(
301        match_state: &mut Match,
302        target_board: &mut PlayerBoard,
303    ) -> Result<String, GameError> {
304        let x = match_state.pending_x
305            .ok_or_else(|| GameError::Invalid("no pending shot"))?;
306        let y = match_state.pending_y
307            .ok_or_else(|| GameError::Invalid("no pending shot"))?;
308        let shooter = match_state.pending_shooter.as_ref()
309            .ok_or_else(|| GameError::Invalid("no pending shot"))?;
310
311        let cur = target_board.get_board().get(BOARD_SIZE, x, y);
312        let is_hit = matches!(cur, Cell::Ship);
313
314        if is_hit {
315            target_board.get_board_mut().set(BOARD_SIZE, x, y, Cell::Hit);
316            target_board.decrement_ships();
317            
318            if target_board.get_ship_count() == 0 {
319                match_state.set_winner(shooter.clone());
320            }
321        } else {
322            target_board.get_board_mut().set(BOARD_SIZE, x, y, Cell::Miss);
323        }
324
325        Ok(match_state.resolve_shot(is_hit))
326    }
327}
